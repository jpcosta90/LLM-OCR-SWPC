% Slide 1
\begin{frame}{Title and Motivation}
\Large
\textbf{Cognitive and Visual Efficiency of Hanzi in Multimodal AI:}\newline
\textbf{Theoretical Foundation for LLM--OCR}
\vspace{0.4cm}
\begin{itemize}
    \item Hanzi as a \textit{cognitively optimized multimodal language}.
    \item Bridging Chinese linguistic theory with modern OCR and LLM architectures.
    \item From \textbf{recognition} to \textbf{understanding}.
\end{itemize}
\end{frame}

% Slide 2
\begin{frame}{From Typewriter to OCR}
\begin{itemize}
    \item The Chinese typewriter pioneered component-based encoding.
    \item Input systems like Wubi and Cangjie evolved from cognitive logic.
    \item Transition from physical encoding to cognitive compression.
\end{itemize}
\includegraphics[width=0.8\textwidth]{typewriter_ocr.png}
\end{frame}

% Slide 3
\begin{frame}{Theoretical Basis: Once-Learning and SWPC}
\begin{itemize}
    \item \textbf{Once-Learning:} One-shot generalization mechanism.
    \item \textbf{SWPC:} Six-Writings Multimodal Coding (形--声--义).
    \item Hanzi = internal multimodality (semantic, phonetic, graphic).
\end{itemize}
\includegraphics[width=0.75\textwidth]{swpc_pyramid.png}
\end{frame}

% Slide 4
\begin{frame}{Research Hypothesis}
\begin{itemize}
    \item Hanzi achieves higher visual-semantic efficiency than Kana or Latin scripts.
    \item DeepSeek-OCR achieves 10× visual compression with 97\% decoding accuracy.
    \item Hypothesis: Hanzi’s pictosemantic coupling drives this efficiency.
\end{itemize}
\includegraphics[width=0.8\textwidth]{ocr_compression.png}
\end{frame}

% Slide 5
\begin{frame}{Methodology Overview}
\begin{itemize}
    \item Step 1: Image-to-text decoding (DeepSeek-OCR).
    \item Step 2: Text re-encoding via SWPC decomposition.
    \item Step 3: Entropy-based evaluation (ABL, VTD, SDA, $\Delta H$).
\end{itemize}
\includegraphics[width=0.9\textwidth]{workflow_pipeline.png}
\end{frame}

% Slide 6
\begin{frame}{Data and Metrics}
\begin{itemize}
    \item Dataset: Periodic Table elements (e.g., 氢–H–Hydrogen).
    \item Metrics:
    \begin{itemize}
        \item ABL – Average Byte Length
        \item VTD – Visual Token Density
        \item SDA – Semantic Decoding Accuracy
        \item $\Delta H$ – Entropy Shift
    \end{itemize}
\end{itemize}
\includegraphics[width=0.8\textwidth]{data_table.png}
\end{frame}

% Slide 7
\begin{frame}{Experimental Results}
\begin{itemize}
    \item Hanzi preserves 97\% semantic recovery under compression.
    \item One-third byte length of Kana/English.
    \item SWPC encoding improves multimodal stability.
\end{itemize}
\includegraphics[width=0.85\textwidth]{results_chart.png}
\end{frame}

% Slide 8
\begin{frame}{Discussion}
\begin{itemize}
    \item CogVLM: external multimodality (vision + text).
    \item Hanzi: internal multimodality (form + sound + meaning).
    \item Our work explains \textit{why} OCR models excel with Hanzi.
    \item Cognitive shift from recognition to understanding.
\end{itemize}
\includegraphics[width=0.8\textwidth]{comparison_cogvlm_swpc.png}
\end{frame}

% Slide 9
\begin{frame}{Conclusion and Outlook}
\begin{itemize}
    \item Hanzi demonstrates cognitive and visual efficiency for multimodal AI.
    \item Once-Learning and SWPC form a consistent theoretical foundation.
    \item Future LLM--OCR systems can leverage Hanzi’s multimodal structure.
\end{itemize}

\end{frame}
